1function out = xtc_processor_func(in, params)
 2% xtc_processor_func - DSP core for the Crosstalk plugin.
 3% This version uses a functional approach with persistent variables,
 4% making it compatible with older versions of MATLAB Coder like R2017b.
 5
 6% %% CAMBIO CLAVE: Declarar los filtros como variables persistentes.
 7% Estas variables se crearán una sola vez y mantendrán su estado
 8% entre las llamadas a esta función.
 9nSamplesIR=8192;
10persistent pFilter_LL pFilter_RL pFilter_LR pFilter_RR isInitialized
11
12% On the very first run, or if the sample rate changes, initialize.
13if isempty(isInitialized)
14    % Crear los objetos de filtro por primera vez
15    pFilter_LL = dsp.FrequencyDomainFIRFilter;
16    pFilter_RL = dsp.FrequencyDomainFIRFilter;
17    pFilter_LR = dsp.FrequencyDomainFIRFilter;
18    pFilter_RR = dsp.FrequencyDomainFIRFilter;
19
20    % Calcular la IR inicial con los parámetros por defecto
21    [cLL, cLR] = transaural_ir3(...
22        params.D, params.dp, params.do, params.beta, nSamplesIR, params.SampleRate);
23    
24    % Asignar los numeradores a los filtros
25    pFilter_LL.Numerator = cLL.';
26    pFilter_RL.Numerator = cLR.';
27    pFilter_LR.Numerator = cLR.';
28    pFilter_RR.Numerator = cLL.';
29    
30    isInitialized = true;
31end
32
33% --- Actualización de Parámetros (si es necesario) ---
34% En un plugin real, esto se llamaría solo cuando un parámetro cambia.
35% Aquí, para la generación de código, asumimos que 'params' puede cambiar
36% en cada llamada, así que recalculamos.
37[cLL, cLR] = transaural_ir3(...
38    params.D, params.dp, params.do, params.beta, nSamplesIR, params.SampleRate);
39
40pFilter_LL.Numerator = cLL.';
41pFilter_RL.Numerator = cLR.';
42pFilter_LR.Numerator = cLR.';
43pFilter_RR.Numerator = cLL.';
44
45% --- Procesamiento de Audio ---
46left_in = in(:,1);
47right_in = in(:,2);
48
49% Usar 'step' en los objetos de filtro persistentes
50y_LL = step(pFilter_LL, left_in);
51y_RL = step(pFilter_RL, right_in);
52y_LR = step(pFilter_LR, left_in);
53y_RR = step(pFilter_RR, right_in);
54
55% Combinar las salidas
56outL = y_LL + y_RL;
57outR = y_LR + y_RR;
58
59out = [outL, outR];
60end
