129function checkForTampering(plugin, paramState, sampleRate, cause)
130% Verify parameters were not tampered with
131if ~isequal(paramState.D, plugin.D)
132    error('ValidateAudioPlugin:ParamChanged', ...
133        '%s changed parameter ''D'' from %g to %g.', ...
134        cause, paramState.D, plugin.D);
135end
136if ~isequal(paramState.dp, plugin.dp)
137    error('ValidateAudioPlugin:ParamChanged', ...
138        '%s changed parameter ''dp'' from %g to %g.', ...
139        cause, paramState.dp, plugin.dp);
140end
141if ~isequal(paramState.do, plugin.do)
142    error('ValidateAudioPlugin:ParamChanged', ...
143        '%s changed parameter ''do'' from %g to %g.', ...
144        cause, paramState.do, plugin.do);
145end
146if ~isequal(paramState.beta, plugin.beta)
147    error('ValidateAudioPlugin:ParamChanged', ...
148        '%s changed parameter ''beta'' from %g to %g.', ...
149        cause, paramState.beta, plugin.beta);
150end
151if ~isequal(paramState.loadIR_Enum, plugin.loadIR_Enum)
152    error('ValidateAudioPlugin:ParamChanged', ...
153        '%s changed parameter ''loadIR_Enum'' from ''%s'' to ''%s'.'', ...
154        cause, paramState.loadIR_Enum, plugin.loadIR_Enum);
155end
156% Verify sample rate was not tampered with
157if ~isequal(getSampleRate(plugin), sampleRate)
158    error('ValidateAudioPlugin:SampleRateChanged', ...
159        '%s changed sample rate from %g to %g.', ...
160        cause, sampleRate, getSampleRate(plugin));
161end
162end

164function y = logchirp(f0, f1, Fs, nsamples, initialPhase)
165% logarithmically swept sine from f0 to f1 over nsamples, at Fs
166y = zeros(nsamples,numel(initialPhase));
167instPhi = logInstantaneousPhase(f0, f1, Fs, nsamples);
168for i = 1:numel(initialPhase)
169    y(:,i) = sin(instPhi + initialPhase(i));
170end
171end
172
173function phi = logInstantaneousPhase(f0, f1, Fs, n)
174final = n-1;
175t = (0:final)/final;
176t1 = final/Fs;
177phi = 2*pi * t1/log(f1/f0) * (f0 * (f1/f0).^(t') - f0);
178end
