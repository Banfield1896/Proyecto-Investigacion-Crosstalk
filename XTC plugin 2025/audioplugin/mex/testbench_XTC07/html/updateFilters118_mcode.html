1classdef XTC07 < audioPlugin
  2    % XTC05 - Crosstalk Cancellation Audio Plugin
  3    % Versión final que utiliza 4 filtros para un procesamiento de estado correcto.
  4
  5    properties
  6        D = 1.0;
  7        dp = 0.2;
  8        do = 0.18;
  9        beta = 0.01;
 10        loadIR_Enum = 'No';
 11    end
 12    
 13    properties(Access = private)
 14        % %% CAMBIO CLAVE: Se necesitan 4 filtros, uno para cada ruta de la señal.
 15        % Por simetría, LL=RR y LR=RL, pero se necesitan instancias separadas
 16        % porque procesan diferentes entradas (canal L vs canal R).
 17        pFilter_LL; % In_L -> Out_L
 18        pFilter_RL; % In_R -> Out_L
 19        pFilter_LR; % In_L -> Out_R
 20        pFilter_RR; % In_R -> Out_R
 21        
 22        nSamplesIR = 8192;
 23        pWavIR=[];
 24    end
 25    
 26    properties (Constant)
 27        PluginInterface = audioPluginInterface(...
 28            audioPluginParameter('D', 'DisplayName', 'Distancia Altavoces-Oídos (m)', 'Mapping', {'lin', 0.5, 2}), ...
 29            audioPluginParameter('dp', 'DisplayName', 'Distancia entre Altavoces (m)', 'Mapping', {'lin', 0.1, 1}), ...
 30            audioPluginParameter('do', 'DisplayName', 'Distancia entre Oídos (m)', 'Mapping', {'lin', 0.1, 0.3}), ...
 31            audioPluginParameter('beta', 'DisplayName', 'Regularización ?', 'Mapping', {'log', 1e-4, 1e-1}),...
 32            audioPluginParameter('loadIR_Enum', 'DisplayName', 'Cargar IR (WAV)', 'Mapping', {'enum', 'No','Sí'}), ...
 33            'InputChannels', 2, ...
 34            'OutputChannels', 2, ...
 35            'PluginName', 'CrosstalkConvolver' ...
 36            );
 37    end
 38    
 39    methods
 40        function plugin = XTC07()
 41            plugin.updateFilters();
 42        end
 43        
 44        function out = process(plugin, in)
 45            if isempty(plugin.pFilter_LL) % Chequeo de seguridad
 46                out = in;
 47                return;
 48            end
 49            
 50            left_in = in(:,1);
 51            right_in = in(:,2);
 52            
 53            % %% CAMBIO CLAVE: Lógica de procesamiento correcta usando 4 filtros.
 54            % Cada filtro se llama UNA SOLA VEZ por bloque.
 55            y_LL = plugin.pFilter_LL(left_in);
 56            y_RL = plugin.pFilter_RL(right_in);
 57            
 58            y_LR = plugin.pFilter_LR(left_in);
 59            y_RR = plugin.pFilter_RR(right_in);
 60            
 61            % Sumar las salidas para cada canal
 62            outL = y_LL + y_RL;
 63            outR = y_LR + y_RR;
 64            
 65            out = [outL, outR];
 66        end
 67        
 68        function reset(plugin)
 69            % Reiniciar los 4 filtros
 70            if ~isempty(plugin.pFilter_LL)
 71                reset(plugin.pFilter_LL);
 72                reset(plugin.pFilter_RL);
 73                reset(plugin.pFilter_LR);
 74                reset(plugin.pFilter_RR);
 75            end
 76        end
 77        
 78        % --- MÉTODOS 'SET' PARA ACTUALIZAR PARÁMETROS ---
 79        function set.D(plugin, val)
 80            plugin.D = val;
 81            plugin.updateFilters();
 82        end
 83        function set.dp(plugin, val)
 84            plugin.dp = val;
 85            plugin.updateFilters();
 86        end
 87        function set.do(plugin, val)
 88            plugin.do = val;
 89            plugin.updateFilters();
 90        end
 91        function set.beta(plugin, val)
 92            plugin.beta = val;
 93            plugin.updateFilters();
 94        end
 95        function set.loadIR_Enum(plugin, val)
 96            plugin.loadIR_Enum = val;
 97            if strcmp(val, 'Sí')
 98                plugin.loadWavIR(); 
 99            end
100            plugin.updateFilters();
101        end
102    end
103
104    methods (Access = private)

105         function updateFilters(plugin)
106             fs = getSampleRate(plugin);
107             if fs == 0, return; end
108             
109             % Obtener las respuestas al impulso (ipsilateral cLL y contralateral cLR)
110             if strcmp(plugin.loadIR_Enum, 'Sí') && ~isempty(plugin.pWavIR)
111                 ir_data = plugin.pWavIR;
112                 len = min(plugin.nSamplesIR, size(ir_data, 1));
113                 cLL = zeros(plugin.nSamplesIR, 1);
114                 cLR = zeros(plugin.nSamplesIR, 1);
115                 cLL(1:len) = ir_data(1:len, 1);
116                 cLR(1:len) = ir_data(1:len, 2);
117             else
118                 [cLL, ~, cLR, ~, ~, ~, ~, ~] = transaural_ir2(...
119                     plugin.D, plugin.dp, plugin.do, plugin.beta, plugin.nSamplesIR, fs);
120             end            
121                         
122             % %% CAMBIO CLAVE: Crear las 4 instancias de filtro
123             % Filtro para In_L -> Out_L (usa la IR ipsilateral cLL)
124             plugin.pFilter_LL = dsp.FrequencyDomainFIRFilter(...
125                'Numerator', cLL.');
126            
127             % Filtro para In_R -> Out_L (usa la IR contralateral cLR)
128             plugin.pFilter_RL = dsp.FrequencyDomainFIRFilter(...
129                'Numerator', cLR.');
130                
131             % Filtro para In_L -> Out_R (usa la IR contralateral cLR)
132             plugin.pFilter_LR = dsp.FrequencyDomainFIRFilter(...
133                'Numerator', cLR.');
134
135             % Filtro para In_R -> Out_R (usa la IR ipsilateral cLL)
136             plugin.pFilter_RR = dsp.FrequencyDomainFIRFilter(...
137                'Numerator', cLL.');
138         end
139         

140         function loadWavIR(plugin)
141             [filename, pathname] = uigetfile('*.wav', 'Seleccionar Archivo de Respuesta al Impulso (Estéreo)');        
142             if ischar(filename)
143                 filepath = fullfile(pathname, filename);            
144                 try
145                     [ir_data, Fs] = audioread(filepath);                
146                     if size(ir_data, 2) ~= 2
147                         warning('El archivo WAV debe ser estéreo.');
148                         plugin.pWavIR = []; return;
149                     end
150                     if Fs ~= getSampleRate(plugin)
151                         warning('Fs del WAV no coincide. Remuestreando...');
152                         ir_data = resample(ir_data, getSampleRate(plugin), Fs);
153                     end
154                     % Primero la limpiamos por si había una IR anterior.
155                     plugin.pWavIR(:) = 0; 
156                     % Calculamos cuántas muestras copiar (para no exceder el tamaño)
157                     lenToCopy = min(size(ir_data, 1), plugin.nSamplesIR);
158                     % Copiamos los datos
159                     plugin.pWavIR(1:lenToCopy, :) = ir_data(1:lenToCopy, :);
160                     disp('IR cargada correctamente desde el archivo WAV.');
161                 catch ME
162                     warning(['Error al cargar el archivo: ', ME.message]);
163                     plugin.pWavIR = [];
164                 end
165             else
166                 plugin.loadIR_Enum = 'No';
167             end
168             plugin.updateFilters();
169         end         
170    end    
171end
